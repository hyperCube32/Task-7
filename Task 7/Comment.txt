int Arr(int arrLength)
{
    srand(time(0));
    int* array = new int[arrLength + 2]; // comment3
    for (int i = 1; i < arrLength + 1; ++i)
    {
        array[i] = rand() % 1000;
    }
    cout << "Массив = ";
    for (int i = 1; i < arrLength + 1; ++i)//commet4
    {
        cout << array[i] << "  ";
    }
    cout << endl;
    int min = 0;
    for (int i = 1; i < arrLength + 1; ++i)
    {
        if ((array[i - 1] > array[i]) && (array[i + 1] > array[i]) && (array[i] > min))
        {
            min = array[i];
        }
    }
    delete[] array;
    return min;
}
